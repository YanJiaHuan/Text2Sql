SELECT AVG(singer.Age) AS average_age, MIN(singer.Age) AS minimum_age, MAX(singer.Age) AS maximum_age FROM singer JOIN singer_in_concert ON singer.Singer_ID = singer_in_concert.Singer_ID JOIN concert ON concert.concert_ID = singer_in_concert.concert_ID JOIN stadium ON concert.Stadium_ID = stadium.Stadium_ID WHERE singer.Country = 'France'

SELECT singer.Name, MIN(singer.Age) AS youngest_age, singer.Song_Name, singer.Song_release_year FROM singer JOIN singer_in_concert ON singer.Singer_ID = singer_in_concert.Singer_ID JOIN concert ON concert.concert_ID = singer_in_concert.concert_ID GROUP BY singer.Singer_ID ORDER BY youngest_age ASC LIMIT 1
